 import numpy as  np

  

def  simulate_attack(num_dice):
    """
    takes num_dice as input and returns True if the attacker wins
    """
    attacker_rolls  = [np.random.choice([1, 2, 3, 4, 5, 6]) for  i  in  range(num_dice)]
    defender_roll  =  np.random.choice([1,2,3,4,5,6])
    return  max(attacker_rolls) >  defender_roll

num_sims  =  10000
num_attacker_rolls  = [1,2,3]
results  = {}
for  num_dice  in  num_attacker_rolls:
    results[num_dice] =  sum([simulate_attack(num_dice) for  i  in  range(num_sims)])/num_sims
	
print("======================================|\nResults: |\n______________________________________|\n||# attacker dice | pct attacker win ||")
for  result  in  results.keys():
    results[result] =  np.around(100*results[result], 4)
    num_spaces_at_end  = (5-len(str(results[result]))) +  11
    print(f"| {result} | {results[result]}%{' '*num_spaces_at_end}||")
    #print("|_________________|___________________||")
    print("|====================================||")
